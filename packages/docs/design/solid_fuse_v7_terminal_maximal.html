FUSECounter.tsx15 ln
import { createSignal } from "solid-js";
import { Column, Text, Button } from "solid-fuse";

export default function Counter() {
  const [count, setCount] = createSignal(0);

  return (
    <Column mainAxisAlignment="center">
      <Text style="headlineMedium">
        Tapped {count()} times
      </Text>
      <Button onPressed={() => setCount(c => c + 1)}>
        Increment
      </Button>
    </Column>
  );
}

import 'package:flutter/material.dart';

class Counter extends StatefulWidget {
  const Counter({super.key});
  @override
  State<Counter> createState() =>
    _CounterState();
}

class _CounterState
    extends State<Counter> {
  int _count = 0;

  @override
  Widget build(BuildContext ctx) {
    return Column(
      mainAxisAlignment:
        MainAxisAlignment.center,
      children: [
        Text(
          'Tapped $_count times',
          style: Theme.of(ctx)
            .textTheme.headlineMedium,
        ),
        ElevatedButton(
          onPressed: () {
            setState(() => _count++);
          },
          child: const Text(
            'Increment'),
        ),
      ],
    );
  }
}
